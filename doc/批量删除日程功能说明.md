```cpp
// 切换日程选择状态
void ScheduleListDialog::toggleScheduleSelection(int scheduleId, bool selected) {
    if (selected) {
        m_selectedScheduleIds.insert(scheduleId);
    } else {
        m_selectedScheduleIds.remove(scheduleId);
    }
    updateBatchDeleteButton();
}

// 更新批量删除按钮状态
void ScheduleListDialog::updateBatchDeleteButton() {
    int count = m_selectedScheduleIds.size();
    m_selectionCountLabel->setText(QString("已选择 %1 项").arg(count));
    m_batchDeleteButton->setEnabled(count > 0);
    
    if (count > 0) {
        m_batchDeleteButton->setText(QString("🗑 批量删除 (%1)").arg(count));
    } else {
        m_batchDeleteButton->setText("🗑 批量删除");
    }
}
```

#### 全选/取消全选

```cpp
// 全选所有日程
void ScheduleListDialog::onSelectAll() {
    QVector<Schedule> allSchedules = dbManager->getSchedules();
    for (const Schedule& schedule : allSchedules) {
        m_selectedScheduleIds.insert(schedule.id);
    }
    updateBatchDeleteButton();
    loadSchedules();
}

// 取消全选
void ScheduleListDialog::onDeselectAll() {
    m_selectedScheduleIds.clear();
    updateBatchDeleteButton();
    loadSchedules();
}
```

#### 批量删除流程

```cpp
void ScheduleListDialog::onBatchDelete() {
    // 1. 检查是否有选中项
    if (m_selectedScheduleIds.isEmpty()) {
        QMessageBox::warning(this, "提示", "请先选择要删除的日程");
        return;
    }
    
    // 2. 确认删除
    int count = m_selectedScheduleIds.size();
    QMessageBox::StandardButton reply = QMessageBox::question(
        this,
        "确认删除",
        QString("确定要删除选中的 %1 项日程吗？\n\n此操作不可撤销！").arg(count),
        QMessageBox::Yes | QMessageBox::No,
        QMessageBox::No
    );
    
    if (reply != QMessageBox::Yes) {
        return;
    }
    
    // 3. 显示加载状态
    showLoadingState(true);
    
    // 4. 延迟执行删除（确保UI更新）
    QTimer::singleShot(100, this, [this, count]() {
        // 执行删除操作
        bool allSuccess = true;
        int deletedCount = 0;
        
        for (int scheduleId : m_selectedScheduleIds) {
            if (dbManager->deleteSchedule(scheduleId)) {
                deletedCount++;
            } else {
                allSuccess = false;
            }
        }
        
        // 5. 隐藏加载状态
        showLoadingState(false);
        
        // 6. 显示结果反馈
        showDeleteFeedback(allSuccess, deletedCount);
        
        // 7. 清空选择并刷新列表
        m_selectedScheduleIds.clear();
        updateBatchDeleteButton();
        loadSchedules();
    });
}
```

#### 加载状态管理

```cpp
void ScheduleListDialog::showLoadingState(bool show) {
    if (show) {
        // 设置覆盖层大小
        m_loadingOverlay->setGeometry(0, 0, this->width(), this->height());
        m_loadingOverlay->raise();
        m_loadingOverlay->show();
        m_loadingLabel->setText("正在删除日程...");
        this->setDisabled(true);  // 禁用对话框交互
    } else {
        m_loadingOverlay->hide();
        this->setEnabled(true);    // 恢复对话框交互
    }
}
```

#### 结果反馈

```cpp
void ScheduleListDialog::showDeleteFeedback(bool success, int count) {
    if (success) {
        QMessageBox::information(
            this,
            "删除成功",
            QString("已成功删除 %1 项日程！").arg(count)
        );
    } else {
        QMessageBox::warning(
            this,
            "部分删除失败",
            QString("成功删除 %1 项日程，但有部分日程删除失败。\n请稍后重试。").arg(count)
        );
    }
}
```

## 交互流程

### 场景1：批量删除多个日程

**步骤**：
1. 用户在日程列表中点击日程项左侧的复选框
2. 复选框变为选中状态（蓝色）
3. 日程项背景变为淡蓝色，显示蓝色边框
4. 工具栏的"选中数量标签"更新显示数量
5. 工具栏的"批量删除"按钮变为可用状态
6. 用户点击"批量删除"按钮
7. 系统弹出确认对话框，显示要删除的数量
8. 用户确认删除
9. 系统显示加载覆盖层，提示"正在删除日程..."
10. 系统执行删除操作
11. 隐藏加载层
12. 显示成功/失败反馈
13. 自动刷新日程列表

### 场景2：全选所有日程

**步骤**：
1. 用户点击工具栏的"☑ 全选"按钮
2. 所有日程项的复选框变为选中状态
3. 工具栏更新显示选中数量
4. "批量删除"按钮显示删除数量

### 场景3：取消全选

**步骤**：
1. 用户点击工具栏的"☐ 取消全选"按钮
2. 所有日程项的复选框变为未选中状态
3. 工具栏显示"已选择 0 项"
4. "批量删除"按钮变为禁用状态
